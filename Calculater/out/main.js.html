/**
 * Division
 * @param {Number} a is first parameter
 * @param {Number} b is first parameter
 * @returns {Object}
 */


function division(a,b){
    // if(!b) throw new Error("Invalid dividend "+b)
    if(!Math.abs(b)) {
        return ("cant divide with 0");
    }
    else if(!Math.abs(a)){
         return 0;
    }
    else {
        return a/b;
    }
}
